# import the JSON utility package
import json
# import the AWS SDK for Python
import boto3
# import time formatting tools
from time import gmtime, strftime

# create a DynamoDB object using the AWS SDK
dynamodb = boto3.resource('dynamodb')
# select your DynamoDB table
table = dynamodb.Table('PowerOfMathDatabase2')
# store current time
now = strftime("%a, %d %b %Y %H:%M:%S +0000", gmtime())

# define the Lambda handler function
def lambda_handler(event, context):
    # extract 'a', 'b', and 'x' from the incoming request
    a = float(event['a'])
    b = float(event['b'])
    x = float(event['x'])
    
    # perform the algebraic calculation: axÂ² + b
    mathResult = a * (x ** 2) + b

    # save result and timestamp to DynamoDB
    response = table.put_item(
        Item={
            'ID': str(mathResult),
            'LatestGreetingTime': now
        })

    # return the result
    return {
        'statusCode': 200,
        'body': json.dumps('Your result is ' + str(mathResult))
    }
